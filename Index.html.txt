<script src="https://cdn.jsdelivr.net/npm/emailjs-com@3/dist/email.min.js"></script>

<script>

emailjs.init("YOUR_PUBLIC_KEY");


let data = {};


function nextPage(id){

document.querySelectorAll(".page").forEach(p=>p.classList.remove("active"));

document.getElementById(id).classList.add("active");

}



function selectFrame(frame){

data.frame = frame;

nextPage("motor");

}



function selectMotor(motor){

data.motor = motor;

nextPage("battery");

}



function saveBattery(){

data.battery = document.getElementById("batterySelect").value;

nextPage("speed");

}



let speedSlider = document.getElementById("speedSlider");

let speedValue = document.getElementById("speedValue");

speedSlider.oninput = function(){

speedValue.innerHTML = this.value + " km/h";

}



function saveSpeed(){

data.speed = speedSlider.value;

nextPage("power");

}



let powerSlider = document.getElementById("powerSlider");

let powerValue = document.getElementById("powerValue");

powerSlider.oninput = function(){

powerValue.innerHTML = this.value + " kW";

}



function savePower(){

data.power = powerSlider.value;

showSummary();

nextPage("form");

}



function showSummary(){

document.getElementById("summary").innerHTML =

"Frame : " + data.frame + "<br>" +

"Motor : " + data.motor + "<br>" +

"Battery : " + data.battery + "<br>" +

"Speed : " + data.speed + " km/h<br>" +

"Power : " + data.power + " kW";

}



function sendMail(){

let templateParams = {

frame:data.frame,

motor:data.motor,

battery:data.battery,

speed:data.speed,

power:data.power,

name:document.getElementById("name").value,

email:document.getElementById("email").value,

message:document.getElementById("message").value

};


emailjs.send("YOUR_SERVICE_ID","YOUR_TEMPLATE_ID",templateParams)

.then(function(){

document.getElementById("sent").innerHTML="Request sent!";

});

}

</script>
